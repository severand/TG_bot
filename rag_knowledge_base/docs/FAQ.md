```python
from fastapi import FastAPI, UploadFile
from rag_module.services import RAGManager
from pathlib import Path
import shutil

app = FastAPI()
manager = RAGManager()

@app.post("/upload")
async def upload_document(file: UploadFile):
    # –°–æ—Ö—Ä–∞–Ω–∏—Ç—å —Ñ–∞–π–ª
    file_path = Path(f"./uploads/{file.filename}")
    with file_path.open("wb") as buffer:
        shutil.copyfileobj(file.file, buffer)
    
    # –î–æ–±–∞–≤–∏—Ç—å –≤ RAG
    doc = manager.add_document(file_path, file.filename)
    
    return {"doc_id": doc.id, "chunks": doc.chunk_count}

@app.get("/search")
async def search(query: str, top_k: int = 5):
    results = manager.search(query, top_k=top_k)
    
    return [
        {
            "text": r.chunk.text,
            "score": r.similarity_score,
            "source": r.source_doc
        }
        for r in results
    ]
```
